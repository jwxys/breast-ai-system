**技术方案**:
```python
# 1. 添加角色检查装饰器
def require_tcm_license(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        current_user = kwargs.get('current_user')
        if not current_user.has_role('TCM_PHYSICIAN'):
            raise HTTPException(403, "需执业中医师资质")
        return f(*args, **kwargs)
    return decorated

# 2. 应用到处方推荐端点
@router.post('/tcm/therapy')
@require_tcm_license
async def recommend_therapy(...):
    ...
```
